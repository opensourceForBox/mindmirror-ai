fun getEmotionDisplay(emotion: Emotion): EmotionDisplay {

        Emotion.CALM -> EmotionDisplay("😌", "平静", Color(0xFF7CB9A8))

### 5.2 连续记录展示

```kotlin
// 连续天数进度条
@Composable
fun StreakProgress(streakDays: Int) {
    val maxDays = 7
    val progress = streakDays.toFloat() / maxDays
    
    Column {
        LinearProgressIndicator(
            progress = progress,
            modifier = Modifier
                .fillMaxWidth()
                .height(4.dp),
            color = MindMirrorColors.primary,
            trackColor = MindMirrorColors.gray100
        )
        Spacer(modifier = Modifier.height(4.dp))
        Text(
            text = "连续记录 $streakDays 天",
            style = MindMirrorTypography.caption,
            color = MindMirrorColors.gray700
        )
    }
}
```

## 6. ViewModel

```kotlin
@HiltViewModel
class HomeViewModel @Inject constructor(
    private val emotionRepository: EmotionRepository,
    private val exerciseRepository: ExerciseRepository,
    private val activityRepository: ActivityRepository
) : ViewModel() {
    
    val uiState: StateFlow<HomeUiState> = _uiState.asStateFlow()
    private val _uiState = MutableStateFlow(HomeUiState())
    
    init {
        loadHomeData()
    }
    
    fun loadHomeData() {
        viewModelScope.launch {
            _uiState.update { it.copy(isLoading = true) }
            
            try {
                // 并行加载数据
                val todayEmotion = emotionRepository.getTodayEmotion()
                val streak = emotionRepository.getStreakDays()
                val recommendedExercises = exerciseRepository.getRecommended(limit = 3)
                val recentActivities = activityRepository.getRecent(limit = 5)
                
                _uiState.update {
                    it.copy(
                        isLoading = false,
                        todayEmotion = todayEmotion,
                        streakDays = streak,
                        recommendedExercises = recommendedExercises,
                        recentActivities = recentActivities
                    )
                }
            } catch (e: Exception) {
                _uiState.update { 
                    it.copy(
                        isLoading = false,
                        error = e.message
                    ) 
                }
            }
        }
    }
    
    fun onRefresh() {
        loadHomeData()
    }
}

data class HomeUiState(
    val isLoading: Boolean = false,
    val todayEmotion: EmotionRecord? = null,
    val streakDays: Int = 0,
    val recommendedExercises: List<Exercise> = emptyList(),
    val recentActivities: List<Activity> = emptyList(),
    val error: String? = null
)
```
